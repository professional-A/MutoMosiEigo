**Tech Stack:** 素のHTML/CSS/JavaScript（ビルドツール・外部ライブラリなし）、Google Fonts（既存と同じ）

<script>
const EXAM_ORDER = ["前期中間試験","前期末試験","後期中間試験","学年末試験"];

let tests = [];
let activeSubject = "すべて";
let activeExam = "すべて";

function getProgress(storageKey, totalItems) {
  try {
    const state = JSON.parse(localStorage.getItem(storageKey) || "{}");
    const done = Object.values(state).filter(v => v && v.done).length;
    return { done, total: totalItems };
  } catch {
    return { done: 0, total: totalItems };
  }
}

function renderFilters() {
  const subjects = ["すべて", ...new Set(tests.map(t => t.subject))];
  const exams = ["すべて", ...EXAM_ORDER.filter(e => tests.some(t => t.exam === e))];

  document.getElementById("subject-filters").innerHTML =
    subjects.map(s =>
      `<button class="fbtn${s === activeSubject ? " active" : ""}" onclick="setSubject('${s}')">${s}</button>`
    ).join("");

  document.getElementById("exam-filters").innerHTML =
    exams.map(e =>
      `<button class="fbtn${e === activeExam ? " active" : ""}" onclick="setExam('${e}')">${e}</button>`
    ).join("");
}

function filtered() {
  return tests.filter(t =>
    (activeSubject === "すべて" || t.subject === activeSubject) &&
    (activeExam === "すべて" || t.exam === activeExam)
  );
}

function renderGroups() {
  const list = filtered();
  const groupMap = new Map();
  list.forEach(t => {
    const key = `${t.year}年度 ${t.grade}年生`;
    if (!groupMap.has(key)) groupMap.set(key, []);
    groupMap.get(key).push(t);
  });

  const sortedKeys = [...groupMap.keys()].sort((a, b) => b.localeCompare(a));

  if (sortedKeys.length === 0) {
    document.getElementById("groups").innerHTML =
      `<div class="empty">該当する模試がありません</div>`;
    return;
  }

  document.getElementById("groups").innerHTML = sortedKeys.map(key => {
    const cards = groupMap.get(key).map(t => {
      const { done, total } = getProgress(t.storageKey, t.totalItems);
      const pct = total ? Math.round(done / total * 100) : 0;
      const untouched = done === 0;
      return `
        <a class="card" href="${t.path}">
          <div class="card-exam">${t.year}年度 ${t.grade}年 ${t.exam}</div>
          <div class="card-subject">${t.subject}</div>
          <div class="progress-bar-wrap">
            <div class="progress-bar" style="width:${pct}%"></div>
          </div>
          <div class="progress-text${untouched ? " untouched" : ""}">
            ${untouched ? "未着手" : `${done} / ${total} 問`}
          </div>
          <div class="card-arrow">開く →</div>
        </a>`;
    }).join("");
    return `<div class="group">
      <div class="group-title">${key}</div>
      <div class="grid">${cards}</div>
    </div>`;
  }).join("");
}

function setSubject(s) { activeSubject = s; renderFilters(); renderGroups(); }
function setExam(e) { activeExam = e; renderFilters(); renderGroups(); }

fetch("tests.json")

## 新しい模試を追加するときの手順（運用メモ）

1. `tests/<年度-学年-試験-科目>/index.html` を作成（既存の模試HTMLをベースに）
2. その模試HTML内の `localStorage.setItem` のキー名をユニークに変更（例: `mutou_2026_4_zenki_eigo2`）
3. `tests.json` に以下のエントリを追加：

```json
{
  "year": 2026,
  "grade": 4,
  "exam": "前期中間試験",
  "subject": "制御工学",
  "path": "tests/2026-4-zenki-chukan-seigyo/",
  "storageKey": "mutou_2026_4_zenki_seigyo",
  "totalItems": 50
}
```

4. git add / commit / push → ホームページに自動反映される
